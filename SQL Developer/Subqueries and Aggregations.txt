Create Students table
CREATE TABLE Students (
    student_id INT PRIMARY KEY,
    name VARCHAR(100),
    math_score INT,
    science_score INT,
    english_score INT
);

Top 5 students based on total scores
SELECT name,
       (math_score + science_score + english_score) AS total_score
FROM Students
ORDER BY total_score DESC
LIMIT 5;

Average math score where math_score > 70
SELECT AVG(math_score) AS average_math_score
FROM Students
WHERE math_score > 70;


Use subquery to filter total score between 200â€“250
SELECT AVG(total) AS avg_total_score
FROM (
    SELECT (math_score + science_score + english_score) AS total
    FROM Students
) AS totals
WHERE total BETWEEN 200 AND 250;


Second highest math score
SELECT MAX(math_score) AS second_highest_math_score
FROM Students
WHERE math_score < (SELECT MAX(math_score) FROM Students);
